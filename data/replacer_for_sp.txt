	return s.replace(/（/g, '(').replace(/）/g, ')').replace(/～/g, '~').replace(/！/g, '!').replace(/”/g, "\"").replace(/ /g, '').replace(/ê/g, 'e').replace(/％/g, '%').replace(/…/g, '...').toLowerCase();
}

// COMMON - json match
function findjson(attrval) {
	for (i_=0; i_<iidxJSONdata.musicdata.length; i_++) {
		if (normalizeStr(iidxJSONdata.musicdata[i_].data['title']) == normalizeStr(attrval)) {
			return iidxJSONdata.musicdata[i_].data['id'];	// return id
		}
	}
	return attrval;// or null
}

// COMMON - dict to tab (output)
// check duplicates
function dict2tab(data) {
	var r = "";
	var d = {};
	for (var k in data) {
		r += k;
		for (var t in data[k]) {
			// title replacement can be done
			r += "\t";
			var id = findjson(data[k][t]);
			r += id;
			if (id in d) {
				console.log(id + " - duplicated");
			}
			d[r] = 0;
		}
		r += "\n";
	}
	return r;
}
